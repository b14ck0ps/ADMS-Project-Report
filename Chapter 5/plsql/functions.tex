1. Calculate Age using a  Function 

\begin{lstlisting}[caption={ Query 1},label={lst:q-1}]
    CREATE OR REPLACE FUNCTION calculate_age(birth_date DATE) RETURN NUMBER IS
    age NUMBER;
BEGIN
    age := TRUNC(MONTHS_BETWEEN(SYSDATE, birth_date) / 12);
    RETURN age;
END;
